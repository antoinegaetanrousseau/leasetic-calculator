'use server';

import { eq, and, isNull, gt, sql } from 'drizzle-orm';
import { auth } from './index';
import { hashToken } from './tokens';
import { setPasswordSchema } from './schemas';
import { db, schema } from '@/lib/db';

export type RedeemKind = 'invite' | 'reset';

export type RedeemResult =
  | { ok: true }
  | { ok: false; reason: 'invalid' | 'invalid_password' | 'server_error' };

/**
 * Atomic token redemption for both invitation (kind='invite') and reset (kind='reset').
 *
 * On success:
 *  - Hashes the new password via Better Auth's configured argon2id hasher
 *  - Upserts the accounts row (creates for invite-flow, updates for reset-flow)
 *  - Marks passwordResets.usedAt = NOW() (single-use enforcement, D-09)
 *  - For kind='reset': bumps users.sessionVersion so requireUser() rejects any
 *    stale sessions on the next request (D-23 mirror — applied here so a user
 *    who reset their password also evicts old sessions)
 *
 * Always returns the same generic 'invalid' reason for any token issue (anti-
 * enumeration sibling of D-22) — the redemption page does NOT distinguish
 * "expired" from "wrong kind" from "missing" in the visible error.
 */
export async function redeemToken(
  plaintext: string,
  kind: RedeemKind,
  password: string,
  confirmPassword: string,
): Promise<RedeemResult> {
  // Input validation via the same Zod schema the client uses
  const parsed = setPasswordSchema.safeParse({ password, confirmPassword });
  if (!parsed.success) return { ok: false, reason: 'invalid_password' };

  try {
    const tokenHash = hashToken(plaintext);

    // Atomic lookup: hash + kind + not-yet-used + not-yet-expired
    const record = await db().query.passwordResets.findFirst({
      where: and(
        eq(schema.passwordResets.tokenHash, tokenHash),
        eq(schema.passwordResets.kind, kind),
        isNull(schema.passwordResets.usedAt),
        gt(schema.passwordResets.expiresAt, new Date()),
      ),
    });
    if (!record) return { ok: false, reason: 'invalid' };

    // Hash the new password using Better Auth's configured hasher.
    // Better Auth exposes auth.context for accessing the password.hash function;
    // if that's not stable, call @node-rs/argon2 directly with the same parameters
    // (memoryCost: 19456, timeCost: 2, parallelism: 1, algorithm: 2 = argon2id).
    const argon2 = await import('@node-rs/argon2');
    const passwordHash = await argon2.hash(password, {
      algorithm: 2,
      memoryCost: 19456,
      timeCost: 2,
      parallelism: 1,
    });

    // Look up the user to scope the accounts row (need email for accountId)
    const user = await db().query.users.findFirst({
      where: eq(schema.users.id, record.userId),
      columns: { id: true, email: true },
    });
    if (!user) return { ok: false, reason: 'invalid' };

    // Upsert accounts row: Better Auth uses providerId='credential' for the
    // email+password account. Generate an id matching Better Auth's nanoid shape
    // for new accounts rows (~21 chars).
    const existingAccount = await db().query.accounts.findFirst({
      where: and(
        eq(schema.accounts.userId, user.id),
        eq(schema.accounts.providerId, 'credential'),
      ),
    });

    if (existingAccount) {
      await db().update(schema.accounts)
        .set({ password: passwordHash, updatedAt: sql`NOW()` })
        .where(eq(schema.accounts.id, existingAccount.id));
    } else {
      // Insert a new accounts row for the credentials provider.
      // The id format mirrors Better Auth's internal nanoid (21 char base64url-ish);
      // randomBytes(16).toString('base64url').slice(0,21) is a valid substitute.
      const { randomBytes } = await import('node:crypto');
      const accountId = randomBytes(16).toString('base64url').slice(0, 21);
      await db().insert(schema.accounts).values({
        id: accountId,
        userId: user.id,
        accountId: user.email,
        providerId: 'credential',
        password: passwordHash,
      });
    }

    // Mark token as used (single-use enforcement, D-09)
    await db().update(schema.passwordResets)
      .set({ usedAt: sql`NOW()` })
      .where(eq(schema.passwordResets.id, record.id));

    // For reset flow: bump sessionVersion to evict any stale sessions
    if (kind === 'reset') {
      await db().update(schema.users)
        .set({ sessionVersion: sql`${schema.users.sessionVersion} + 1` })
        .where(eq(schema.users.id, user.id));
      // Also call Better Auth session-revoke to delete DB sessions immediately
      await auth.api.revokeUserSessions({ body: { userId: user.id } });
    }

    return { ok: true };
  } catch (e) {
    // Bounded error redaction (Phase 5 classifyError discipline): never leak
    // DB schema details or specific failure reasons to the caller.
    console.error('[redeem] token redemption failed:', e);
    return { ok: false, reason: 'server_error' };
  }
}

import { useState, useEffect } from 'react';
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { toast } from 'sonner';
import { useRouter, useSearchParams } from 'next/navigation';
import { authClient } from '@/lib/auth/client';
import { loginSchema, type LoginInput } from '@/lib/auth/schemas';
import { t, type Lang } from '@/lib/i18n';

interface LoginFormProps {
  lang: Lang;
}

export function LoginForm({ lang }: LoginFormProps) {
  const router = useRouter();
  const searchParams = useSearchParams();
  const [serverError, setServerError] = useState<string | null>(null);

  const {
    register, handleSubmit,
    formState: { errors, isSubmitting },
  } = useForm<LoginInput>({ resolver: zodResolver(loginSchema), mode: 'onBlur' });

  // Mount-time success toasts driven by query params (D-09 invitation/reset success
  // flow + D-24 logout flow). Strip the param via router.replace so refresh doesn't re-toast.
  useEffect(() => {
    const invited = searchParams.get('invited');
    const reset = searchParams.get('reset');
    const loggedOut = searchParams.get('logged_out');
    if (invited === '1') toast.success(t('auth.toast.invite.success', lang));
    else if (reset === '1') toast.success(t('auth.toast.reset.success', lang));
    else if (loggedOut === '1') toast.success(t('auth.toast.logout.success', lang));
    if (invited || reset || loggedOut) router.replace('/login');
  }, [searchParams, router, lang]);

  const onSubmit = async (data: LoginInput) => {
    setServerError(null);
    const next = searchParams.get('next');
    const callbackURL = next && next.startsWith('/') ? next : '/';
    const { error } = await authClient.signIn.email({
      email: data.email.toLowerCase(),
      password: data.password,
      callbackURL,
    });
    if (error) {
      // ALWAYS the same generic message regardless of underlying cause (D-22 / AUTH-04)
      setServerError(t('auth.error.invalid.credentials', lang));
    }
    // On success Better Auth navigates via callbackURL; nothing else to do.
  };

  return (

        {t('auth.signin.title', lang)}
      </div>

      {/* Email field */}
      <label htmlFor="login-email" style={{ display: 'block', fontSize: '11.2px', fontWeight: 500, color: 'var(--ink)', marginBottom: 6 }}>
        {t('auth.field.email', lang)}
      </label>
      <input
        id="login-email"
        type="email"
        autoComplete="email"
        placeholder={t('auth.field.email.placeholder', lang)}
        aria-invalid={!!errors.email}
        aria-describedby={errors.email ? 'login-email-error' : undefined}
        className={errors.email ? 'invalid' : ''}
        {...register('email')}
        style={{ width: '100%', padding: '10px 12px', borderRadius: 12, border: '1px solid var(--border)', background: 'var(--surface)', color: 'var(--ink)', marginBottom: errors.email ? 4 : 16 }}
      />
      {errors.email && <div id="login-email-error" style={{ fontSize: '11.2px', fontWeight: 500, color: 'var(--danger)', marginBottom: 16 }} role="alert">{t('auth.error.email.invalid', lang)}</div>}

      {/* Password field */}
      <label htmlFor="login-password" style={{ display: 'block', fontSize: '11.2px', fontWeight: 500, color: 'var(--ink)', marginBottom: 6 }}>
        {t('auth.field.password', lang)}
      </label>
      <input
        id="login-password"
        type="password"
        autoComplete="current-password"
        placeholder={t('auth.field.password.placeholder', lang)}
        aria-invalid={!!errors.password}
        {...register('password')}
        style={{ width: '100%', padding: '10px 12px', borderRadius: 12, border: '1px solid var(--border)', background: 'var(--surface)', color: 'var(--ink)', marginBottom: 16 }}

          {serverError}
        </div>
      )}

      {/* Submit button — .btn-green pill, full width, spinner replaces label */}
      <button type="submit" disabled={isSubmitting} className="btn-green" style={{ width: '100%', borderRadius: 9999, padding: '0.6rem 1.5rem', fontWeight: 600, fontSize: 14, color: '#fff', background: 'var(--gd)', border: 'none', cursor: isSubmitting ? 'not-allowed' : 'pointer', opacity: isSubmitting ? 0.7 : 1, marginBottom: 12 }}>
        {isSubmitting ? t('auth.button.signin.loading', lang) : t('auth.button.signin', lang)}
      </button>

      {/* Forgot-password hint */}
      <div style={{ fontSize: '10.5px', color: 'var(--muted)', lineHeight: 1.5, marginTop: 8 }}>
        {t('auth.hint.forgot.password', lang)}
      </div>
    </form>
  );
}

'use client';

import { useState, useTransition } from 'react';
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { toast } from 'sonner';
import { useRouter } from 'next/navigation';
import { Eye, EyeOff } from 'lucide-react';
import { setPasswordSchema, type SetPasswordInput } from '@/lib/auth/schemas';
import { redeemToken, type RedeemKind } from '@/lib/auth/redeem';
import { t, type Lang } from '@/lib/i18n';

interface SetPasswordFormProps {
  token: string;
  kind: RedeemKind;
  lang: Lang;
}

function strength(pwd: string): 0 | 1 | 2 | 3 | 4 {
  if (!pwd) return 0;
  let score = 0;
  if (pwd.length >= 8) score++;
  if (/\d/.test(pwd)) score++;
  if (/[A-Z]/.test(pwd)) score++;
  if (pwd.length >= 12 && /[^A-Za-z0-9]/.test(pwd)) score++;
  // UI-SPEC §Invite: ≥8 + number + upper = "Strong" (3); add length+symbol = "Very strong" (4)
  return Math.min(4, score) as 0 | 1 | 2 | 3 | 4;
}
