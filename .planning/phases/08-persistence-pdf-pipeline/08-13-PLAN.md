```tsx
import { getLatestGlobalParams, getProposalById } from '@/lib/db/queries';
import { DuplicatePrefillToast } from '@/components/proposals/DuplicatePrefillToast';
import type { ProposalInput } from '@/lib/calc';

interface PageParams {
  searchParams: Promise<{ duplicate?: string }>;
}

export default async function NewProposalPage({ searchParams }: PageParams) {
  const { session } = await requireUser();
  const lang = await getCurrentLang();
  const sp = await searchParams;

  const params = await getLatestGlobalParams();
  const coefficientsExpired = params === null;

  const u = session.user as { email: string; displayName?: string | null; name?: string | null };
  const partnerName = u.displayName ?? u.name ?? '';

  // Default partial prefill (partnerName only — Plan 07-05 D-7-13 baseline).
  let prefill: Partial<ProposalInput> = { partnerName };

  // PROP-21: ?duplicate=<id> — pre-populate ALL 14 fields from the source's inputs.
  if (sp.duplicate) {
    const source = await getProposalById(sp.duplicate);
    if (source && source.userId === session.user.id) {
      const sourceInputs = source.inputs as Record<string, unknown>;
      // Spread the source inputs, then overlay partnerName from session (so duplicating
      // someone else's commercial name doesn't accidentally happen — though our query
      // already enforces same-user, defence-in-depth at the field level).
      prefill = {
        ...(sourceInputs as Partial<ProposalInput>),
        partnerName,    // session-derived overlay
      };
    }
    // If source missing / not owned → silent fallback to default prefill.
    // NO 404 here — duplicate is a UX nicety, not a critical path.
  }

  return (
    <div>
      {sp.duplicate && <DuplicatePrefillToast lang={lang} />}

      <h1 style={{ /* existing */ }}>{t('header.proposals.new', lang)}</h1>

      <ProposalFormProvider prefill={prefill}>
        <div style={{ /* existing grid */ }}>
          <ProposalForm lang={lang} />
          <LiveLoyerPreview lang={lang} coefficientsExpired={coefficientsExpired} />
        </div>
      </ProposalFormProvider>
    </div>
  );
}
```

**Create `src/components/proposals/DuplicatePrefillToast.tsx`:**

```tsx
'use client';

import { useEffect, useRef } from 'react';
import { useRouter, useSearchParams } from 'next/navigation';
import { toast } from 'sonner';
import { t, type Lang } from '@/lib/i18n/dictionaries';

export interface DuplicatePrefillToastProps { lang: Lang; }

/**
 * Mounts on /proposals/new?duplicate=<id>; on first render fires the
 * "Fields pre-filled..." info toast and strips the ?duplicate= flag from
 * the URL (so a refresh doesn't re-fire the toast and so the partner sees
 * a cleaner URL after the prefill takes effect).
 *
 * Note: stripping ?duplicate= ALSO means the next form submit's
 * useSearchParams().get('duplicate') in ProposalForm returns null. To
 * preserve the duplicatedFromId for the POST body, ProposalForm captures
 * the URL value via useState lazy init at mount time (see Task 1's edit
 * — the mount-time captured value persists across re-renders even after
 * the URL is cleaned).
 */
export function DuplicatePrefillToast({ lang }: DuplicatePrefillToastProps) {
  const router = useRouter();
  const searchParams = useSearchParams();
  const fired = useRef(false);

  useEffect(() => {
    if (fired.current) return;
    if (searchParams.get('duplicate')) {
      toast.info(t('proposal.toast.duplicate.prefilled', lang));
      fired.current = true;
      const next = new URLSearchParams(Array.from(searchParams.entries()));
      next.delete('duplicate');
      const qs = next.toString();
      router.replace(qs.length > 0 ? `?${qs}` : '?', { scroll: false });
    }
  }, [searchParams, router, lang]);

  return null;
}
```

| Boundary | Description |
|----------|-------------|
| ?duplicate=<uuid> → server fetch | Server-side ownership-check on getProposalById; failure silent → empty form. |
| duplicatedFromId in POST body | Plan 08-07 ALREADY accepts this field. Plan 08-13 just supplies it. The server-side validation is at the schema level (string check); ownership of duplicatedFromId itself is NOT re-verified server-side because (a) it's an audit hint, not an authorization input, and (b) the prefill ownership check ALREADY filtered out non-owned IDs at the page-render layer. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-13-01 | Information Disclosure | URL ?duplicate=<other-user's-id> via guess | mitigate | The page server-side ownership-checks; if not owned, prefill silently falls back. No data leaked. |
| T-08-13-02 | Tampering | Client crafts POST body with duplicatedFromId pointing at another user's row | mitigate | The Plan 08-07 server route does NOT verify ownership of duplicatedFromId (audit hint only — accepted in Plan 08-07's threat model T-08-07-05). The new row's data is the attacker's own (they typed it); duplicatedFromId is just a (possibly fake) audit trail. Phase 9 admin UI may surface this hint; PR review for Phase 9 should add ownership verification if commission-pivot info could leak via the admin viewer. Documented in Plan 08-07 SUMMARY. |
| T-08-13-03 | Tampering | Source proposal mutated between detail-page render and the prefill fetch | accept | Race condition is benign — proposals are immutable post-creation (PROP-23); the only mutation is soft-delete which still allows getProposalById to read the inputs. Worst case: a deleted source's inputs are used as prefill, a fresh proposal is created with current global_params. Acceptable. |
| T-08-13-04 | Tampering | URL flag re-fires on browser back-button | mitigate | DuplicatePrefillToast's `fired.current` ref + the URL strip on first mount prevent re-firing. Browser back from /proposals/{newId} → /proposals/new (fresh mount, no ?duplicate=) → no toast. |
