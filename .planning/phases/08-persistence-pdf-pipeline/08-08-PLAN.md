```ts
import { NextResponse, type NextRequest } from 'next/server';
import { requireUser } from '@/lib/auth/require';
import { getProposalById } from '@/lib/db/queries';
import { storage } from '@/lib/storage';

// PROP-13: stream route, NOT raw blob URL.
// Node runtime — storage adapter uses Node fetch / aws-sdk on the s3 path.
export const runtime = 'nodejs';
export const dynamic = 'force-dynamic';

interface RouteParams { params: Promise<{ id: string }> }

export async function GET(_req: NextRequest, ctx: RouteParams) {
  // Step 1: auth.
  let userId: string;
  try {
    const { session } = await requireUser();
    userId = session.user.id;
  } catch {
    return NextResponse.json({ error: 'unauthorized' }, { status: 401 });
  }

  // Step 2: read the proposal row.
  const { id } = await ctx.params;
  const proposal = await getProposalById(id);

  // Step 3: 404 obscurity for not-found OR not-owned (D-18).
  if (!proposal || proposal.userId !== userId) {
    return NextResponse.json({ error: 'not_found' }, { status: 404 });
  }

  // Step 4: 410 if hard-purgeable (deleted_at > 30d). Plan 08-14's CLI MAY
  // already have purged the blob; the row check fails-fast before we hit
  // the storage adapter and 404.
  if (proposal.deletedAt && Date.now() - proposal.deletedAt.getTime() > 30 * 24 * 3600 * 1000) {
    return NextResponse.json({ error: 'gone' }, { status: 410 });
  }

  // Step 5: 404 if no PDF yet (D-B1 tombstone OR mid-flight). Treat as
  // "not-found" since the row has no consumable artifact.
  if (!proposal.pdfBlobKey) {
    return NextResponse.json({ error: 'not_found' }, { status: 404 });
  }

  // Step 6: stream the bytes.
  let blob;
  try {
    blob = await storage().get(proposal.pdfBlobKey);
  } catch (err) {
    console.error('[GET /api/proposals/[id]/pdf] storage.get failed', err);
    return NextResponse.json({ error: 'storage_unavailable' }, { status: 502 });
  }

  return new NextResponse(blob.body, {
    status: 200,
    headers: {
      'Content-Type': blob.contentType || 'application/pdf',
      'Content-Length': String(blob.size),
      'Content-Disposition': `inline; filename="Leasetic_Proposition_${proposal.lcRef}.pdf"`,
      'Cache-Control': 'private, max-age=0, no-store',
    },
  });
}
```

import 'server-only';
import {
  listProposalsByUser, searchProposals,
  type ListResult, encodeCursor, decodeCursor,
} from '@/lib/db/queries';

export interface ProposalRowDto {
  id: string;
  lcRef: string;
  clientCo: string;          // pulled from inputs jsonb
  amountHT: string;          // digit-only, formatted client-side via formatCurrency
  createdAt: string;         // ISO 8601 (server stringifies the Date for JSON wire shape)
  validityDays: 15 | 30 | 60;
  language: 'fr' | 'en';
  deletedAt: string | null;  // present in deleted-view; null otherwise
}

export interface ListResponse {
  rows: ProposalRowDto[];
  hasMore: boolean;
  /** base64url-encoded cursor for the next page; null if no more pages */
  nextCursor: string | null;
}

export interface BuildListParams {
  userId: string;
  q?: string;
  cursorEncoded?: string | null;
  deleted?: boolean;
  limit?: number;
}

/**
 * Used by:
 *   - GET /api/proposals (Plan 08-08 Task 3)
 *   - app/(authed)/page.tsx (Plan 08-11) for the SSR initial render
 *
 * Note: Plan 08-11 calls this directly server-side instead of fetching
 * /api/proposals on the SSR pass — saves one round-trip and keeps the
 * type-safe path. Plan 08-11's "Load More" button is the only client-side
 * caller of GET /api/proposals.
 */
export async function buildListResponse(args: BuildListParams): Promise<ListResponse> {
  const cursor = args.cursorEncoded ? decodeCursor(args.cursorEncoded) : null;
  const q = args.q?.trim() ?? '';

  const result: ListResult = q.length > 0
    ? await searchProposals({
        userId: args.userId, q,
        cursor: cursor ?? undefined,
        deleted: args.deleted ?? false,
        limit: args.limit ?? 20,
      })
    : await listProposalsByUser({
        userId: args.userId,
        cursor: cursor ?? undefined,
        deleted: args.deleted ?? false,
        limit: args.limit ?? 20,
      });

  return {
    rows: result.rows.map((row) => ({
      id: row.id,
      lcRef: row.lcRef,
      clientCo: typeof (row.inputs as { clientCo?: unknown })?.clientCo === 'string'
        ? (row.inputs as { clientCo: string }).clientCo
        : '',
      amountHT: typeof (row.inputs as { amountHT?: unknown })?.amountHT === 'string'
        ? (row.inputs as { amountHT: string }).amountHT
        : '0',
      createdAt: row.createdAt.toISOString(),
      validityDays: ((row.inputs as { validityDays?: unknown })?.validityDays as 15 | 30 | 60) ?? 30,
      language: row.language as 'fr' | 'en',
      deletedAt: row.deletedAt ? row.deletedAt.toISOString() : null,
    })),
    hasMore: result.hasMore,
    nextCursor: result.nextCursor ? encodeCursor(result.nextCursor) : null,
  };
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Stream route → blob storage | The route is the gate. requireUser + ownership before storage().get is called. |
| List endpoint → query helpers | userId from session is the WHERE filter. Even if a malicious client crafts ?cursor=base64-of-other-users-row, the user_id WHERE clause excludes any row not owned by the caller. |
| Cursor decode | Untrusted base64; decodeCursor returns null on malformed → treated as no-cursor. No 400 on decode failure (D-C1: tolerant of stale cookies / shared links). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-08-01 | Information Disclosure | Stream route returns 403 (vs 404) leaking row existence | mitigate | All "no access" cases return 404 (D-18 obscurity). |
| T-08-08-02 | Information Disclosure | Cache-Control header missing → CDN caches private PDF | mitigate | `Cache-Control: private, max-age=0, no-store` set explicitly per PITFALLS §5.4. |
| T-08-08-03 | Information Disclosure | DTO leaks commission_pct from inputs jsonb | mitigate | DTO includes only id/lcRef/clientCo/amountHT/createdAt/validityDays/language/deletedAt. paramsSnapshot.commissionPct is excluded from the wire shape. |
| T-08-08-04 | DoS | List endpoint with limit=10000 | mitigate | The route handler clamps limit to [1, 50]. The helper passes through (defence-in-depth: caller can't unintentionally OOM). |
| T-08-08-05 | Information Disclosure | Cursor encoding is just base64-JSON — predictable | accept | Cursor leakage doesn't grant access; the WHERE clause still filters by user_id. Worst case: a leaked cursor reveals the (createdAt, id) of a row the leaker already saw. |
| T-08-08-06 | DoS | A 404 lookup on a non-existent UUID still queries getProposalById (1 row read) | accept | One indexed PK lookup per request; trivially bounded. Adversary gains no info from response timing differences (Phase 8 doesn't ship constant-time wrapping; v1.2 candidate). |
| T-08-08-07 | Tampering | Future contributor adds an INSECURE shortcut: `if (req.headers.get('x-bypass') === 'admin') skip ownership` | mitigate | requireUser + ownership-check pattern is canonical (Phase 6 D-18). PR review + Phase 9 admin viewer (which will introduce a separate admin-only handler) keep the seam clean. |
| T-08-08-08 | DoS | A 50MB PDF buffered in memory per request | accept | Phase 8 PDFs are <100KB. Phase 11+ may switch to ReadableStream pass-through if size grows. |
