**Step B — Create `src/components/proposals/ValidityChip.tsx`:**

```tsx
import { CircleDot, Clock } from 'lucide-react';
import { t, type Lang } from '@/lib/i18n/dictionaries';
import { formatDate } from '@/lib/i18n/format';

export interface ValidityChipProps {
  createdAt: Date;
  validityDays: 15 | 30 | 60;
  lang: Lang;
}

/**
 * Active vs Expired derivation (D-8-18): renders at render time from
 * (createdAt, validityDays, now). No denormalized column.
 */
export function ValidityChip({ createdAt, validityDays, lang }: ValidityChipProps) {
  const expiresAt = new Date(createdAt.getTime() + validityDays * 86_400_000);
  const isActive = Date.now() < expiresAt.getTime();
  const className = isActive ? 'chip chip-active' : 'chip chip-expired';
  const label = isActive ? t('proposal.chip.active', lang) : t('proposal.chip.expired', lang);
  const tooltipKey = isActive ? 'proposal.chip.tooltip.expires' : 'proposal.chip.tooltip.expired';
  const tooltip = t(tooltipKey, lang).replace('{0}', formatDate(expiresAt, lang));

  return (
    <span className={className} title={tooltip}>
      {isActive ? <CircleDot size={12} fill="currentColor" aria-hidden="true" />
                : <Clock size={12} aria-hidden="true" />}
      {label}
    </span>
  );
}
```

**Create `src/components/proposals/EmbeddedPdfPreview.tsx`:**

```tsx
import { t, type Lang } from '@/lib/i18n/dictionaries';

export interface EmbeddedPdfPreviewProps {
  proposalId: string;
  lang: Lang;
}

/**
 * D-8-10: native <embed> wrapper. Smallest API for inline PDF in Chrome+Edge.
 * Browser-native fallback: if <embed> can't render PDFs (some Firefox configs),
 * the inner fallback link surfaces.
 *
 * Server-renderable (no 'use client') — the route handler /api/proposals/{id}/pdf
 * is the source of bytes; the browser fetches them directly.
 */
export function EmbeddedPdfPreview({ proposalId, lang }: EmbeddedPdfPreviewProps) {
  const src = `/api/proposals/${proposalId}/pdf`;
  return (
    <div className="pdf-embed-wrap" title={t('proposal.detail.pdf.preview.title', lang)}>
      <embed
        src={src}
        type="application/pdf"
        aria-label={t('proposal.detail.pdf.preview.aria', lang)}
      />
      <noscript>
        <a href={src} target="_blank" rel="noopener noreferrer">
          {t('proposal.detail.pdf.fallback.link', lang)}
        </a>
      </noscript>
    </div>
  );
}
```

NOTE: <noscript> is a no-op for the JS path; for the actual fallback (browser doesn't support <embed> for PDFs), browsers replace the embed's render with the embed's children — but JSX `<embed>` is treated as a void element. The cleaner fallback path: render an explicit `<a>` BELOW the embed, visually hidden when embed succeeds, visible only via prefers-reduced-data or browser non-rendering.

For Phase 8 simplicity: `<noscript>` adequately covers the no-JS case; for the actual broken-PDF-embed case (very rare in Chrome+Edge per CLAUDE.md project scope), a future v1.2 enhancement can swap to a more robust fallback (PDF.js or always-visible link). Document the trade-off.

```tsx
                <span style={{
                  fontSize: 24, fontWeight: 700, color: 'var(--navy)',
                  fontVariantNumeric: 'tabular-nums', letterSpacing: '-0.02em',
                }}>
                  {isComputed && computed.loyerHT
                    ? formatCurrency(Number(computed.loyerHT), lang)
                    : t('pdf.loyer.on.demand', lang)}
                </span>
                {isComputed && (
                  <span style={{ fontSize: 13, fontWeight: 400, color: 'var(--muted)' }}>
                    {t('proposal.detail.computed.loyer.suffix', lang)}
                  </span>
                )}
              </div>
            </div>

            {/* Validity / expires line */}
            {(() => {
              const days = (inputs.validityDays as 15 | 30 | 60) ?? 30;
              const expiresAt = new Date(proposal.createdAt.getTime() + days * 86_400_000);
              const isExpired = Date.now() >= expiresAt.getTime();
              const labelKey = isExpired ? 'proposal.detail.computed.expired.label' : 'proposal.detail.computed.expires.label';
              return (
                <dl style={{
                  display: 'grid', gridTemplateColumns: 'max-content 1fr',
                  columnGap: 24, rowGap: 8, alignItems: 'baseline', marginTop: 8,
                }}>
                  <Row k="Validité" v={`${days} ${t('proposal.validity.suffix', lang)}`} />
                  <Row k={t(labelKey, lang).split(' {')[0]} v={formatDate(expiresAt, lang)} />
                </dl>
              );
            })()}
          </section>
        </div>

        {/* Right: PDF preview + action stack (sticky) */}
        <aside style={{ position: 'sticky', top: 'calc(var(--topbar-h) + 24px)' }}>
          <section className="card">
            <div className="ctitle">
              <span>{t('proposal.detail.pdf.preview.title', lang)}</span>
            </div>
            <div style={{ marginTop: 16 }}>
              <EmbeddedPdfPreview proposalId={proposal.id} lang={lang} />
            </div>
          </section>

          <div style={{ display: 'flex', flexDirection: 'column', gap: 8, marginTop: 16 }}>
            {/* Download — same-origin <a download> */}
            <a
              href={`/api/proposals/${proposal.id}/pdf`}
              download={`Leasetic_Proposition_${proposal.lcRef}.pdf`}
              className="btn-green"
              style={{ width: '100%', justifyContent: 'center' }}
            >
              <Download size={17} />
              {t('proposal.detail.action.download', lang)}
            </a>

            {/* Duplicate — Plan 08-13 wires the prefill side; this Link is the entry point */}
            <Link
              href={`/proposals/new?duplicate=${proposal.id}`}
              className="btn-navy"
              style={{ width: '100%', justifyContent: 'center' }}
            >
              <CopyIcon size={17} />
              {t('proposal.detail.action.duplicate', lang)}
            </Link>

            {/* Delete (active) OR Restore (deleted) — Plan 08-12 owns the actual server actions.
                Phase 8-10 ships the BUTTONS pointing at server-action routes 08-12 builds. */}
            {isDeleted ? (
              <RestorePlaceholderButton proposalId={proposal.id} lang={lang} />
            ) : (
              <DeletePlaceholderButton proposalId={proposal.id} lang={lang} />
            )}
          </div>
        </aside>
      </div>
    </div>
  );
}
```
