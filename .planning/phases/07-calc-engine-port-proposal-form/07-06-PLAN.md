```ts
export const dictionaries = {
  fr: {
    'header.title': '...',
    // ... 231 existing keys
  },
  en: {
    'header.title': '...',
    // ... 231 existing keys (must match fr)
  },
} as const;

export type Lang = 'fr' | 'en';
export type DictKey = keyof typeof dictionaries.fr;
export function t(key: DictKey, lang: Lang): string {
  return dictionaries[lang][key] ?? dictionaries.fr[key];
}

// Compile-time parity:
type _EnHasAllFrKeys = {
  [K in DictKey]: K extends keyof typeof dictionaries.en ? true : never;
};
type _EnParityProof = _EnHasAllFrKeys;  // fails compile if any K maps to never
```

NOTE: `t()` does NOT support `{0}` interpolation — values like `'Bonjour, {0}'` are
returned verbatim. Consumers must do their own `.replace('{0}', value)` at the call
site (matches Phase 6 pattern; see header.title.with.partner usage).
