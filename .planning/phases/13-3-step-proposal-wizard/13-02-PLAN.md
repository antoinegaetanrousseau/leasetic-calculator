```ts
/**
 * D-21 edit-invalidates-downstream + D-22/D-23 navigate-preserves-state.
 *
 * @param prevInputs   The draft's CURRENT inputs jsonb (before this update).
 * @param nextInputs   The inputs jsonb about to be persisted.
 * @param fromStep     Which step the partner is leaving (1 | 2).
 *
 * Returns the new _completedSteps array. Algorithm:
 *   1. If no input field changed (only _completedSteps or _uiAccordionOpen changed) → preserve.
 *   2. Else: lowestChangedStep = the smallest step number whose owned inputs differ.
 *   3. Return prevSteps filtered to numbers < lowestChangedStep, then add `fromStep` since the partner is advancing forward through fromStep.
 *
 * Step ownership of inputs:
 *   - Step 1 owns: clientCo, clientName, clientEmail, clientTel, partnerRef, amountHT, durationMonths,
 *     clientRole, clientSiren, projectDesc, slb, evalParc (accordion fields), AND partnerCo + partnerName (session-hydrated)
 *   - Step 2 owns: nothing (read-only per D-11; advancing through step 2 only adds 2 to _completedSteps)
 *   - Step 3 owns: nothing (read-only; clicking Continuer on step 3 = finalize, not advance)
 */
export function deriveCompletedSteps(
  prevInputs: Record<string, unknown>,
  nextInputs: Record<string, unknown>,
  fromStep: 1 | 2,
): number[];

export function markStepCompleted(
  prevSteps: number[] | undefined,
  step: 1 | 2 | 3,
): number[];
```
