  Implementation skeleton:
  ```
  'use client';
  import { useState, useTransition } from 'react';
  import { useRouter } from 'next/navigation';
  import { toast } from 'sonner';
  import { t, type Lang } from '@/lib/i18n/dictionaries';
  import { WizardActionBar } from '@/(authed)/proposals/new/_components/WizardActionBar';

  export function FinalizeButton({ draftId, onSaveDraft, lang }: FinalizeButtonProps) {
    const router = useRouter();
    const [isSubmitting, setIsSubmitting] = useState(false);

    const handleFinalize = async () => {
      setIsSubmitting(true);
      try {
        const res = await fetch('/api/proposals/finalize', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ draftId }),
        });
        if (!res.ok) {
          throw new Error(`finalize_failed_${res.status}`);
        }
        const data = await res.json();
        toast.success(t('wizard.toast.finalize.success', lang));
        router.push(`/proposals/${data.id}`);
      } catch (err) {
        toast.error(t('wizard.toast.finalize.error', lang), { duration: 5000 });
        setIsSubmitting(false);
      }
      // NOTE: on success, do NOT reset isSubmitting — the redirect will unmount this component.
      // Keeping isSubmitting=true prevents the CTA from being re-clickable during the navigation.
    };

    return (
      <WizardActionBar
        currentStep={3}
        draftId={draftId}
        onSaveDraft={onSaveDraft}
        primary={{
          kind: 'action',
          onClick: handleFinalize,
          label: t('wizard.action.step3.confirm', lang),
          spinnerLabel: t('wizard.action.step3.confirm.spinner', lang),
          isSubmitting,
        }}
        lang={lang}
      />
    );
  }
  ```
